HAI_clamp_escalation = {
	HAI = {
		if = {
			limit = { check_variable = { HAI_escalation > 100 } }
			set_variable = { HAI_escalation = 100 }
		}
		if = {
			limit = { check_variable = { HAI_escalation < 0 } }
			set_variable = { HAI_escalation = 0 }
		}
	}
}

HAI_check_escalation_threshold = {
	HAI_clamp_escalation = yes
	HAI = {
		if = {
			limit = {
				has_country_flag = HAI_crisis_active
				check_variable = { HAI_escalation > 99 }
				NOT = { has_country_flag = HAI_war_event_pending }
			}
			set_country_flag = HAI_war_event_pending
			country_event = { id = hai_crisis.20 hours = 6 }
		}
	}
}

# ROOT is the attacking armed group. Haiti receives a national operation lock
# immediately, before the result event is queued, so two AI countries cannot
# launch major attacks in the same hour. The attacker remains exhausted longer
# than the shared eighteen-day window, encouraging natural rotation.
HAI_begin_major_gang_operation = {
	HAI = {
		set_country_flag = HAI_major_gang_operation_active
		country_event = { id = hai_gangs.90 days = 18 }
	}
	add_timed_idea = { idea = HAI_operational_exhaustion days = 45 }
}

HAI_check_regional_influence_threshold = {
	HAI = {
		if = {
			limit = {
				has_country_flag = HAI_regional_contest_active
				NOT = { has_country_flag = HAI_influence_resolution_pending }
				check_variable = { HAI_regional_influence > 59 }
			}
			set_country_flag = HAI_influence_resolution_pending
			if = {
				limit = { has_country_flag = HAI_target_DOM }
				DOM = { country_event = { id = hai_cssm.1 } }
			}
			if = {
				limit = { has_country_flag = HAI_target_JAM }
				JAM = { country_event = { id = hai_cssm.1 } }
			}
			if = {
				limit = { has_country_flag = HAI_target_BAH }
				BAH = { country_event = { id = hai_cssm.1 } }
			}
		}

		if = {
			limit = {
				has_country_flag = HAI_regional_contest_active
				NOT = { has_country_flag = HAI_influence_resolution_pending }
				check_variable = { HAI_regional_influence < -59 }
			}
			set_country_flag = HAI_influence_resolution_pending
			if = {
				limit = { has_country_flag = HAI_target_DOM }
				DOM = { country_event = { id = hai_cssm.2 } }
			}
			if = {
				limit = { has_country_flag = HAI_target_JAM }
				JAM = { country_event = { id = hai_cssm.2 } }
			}
			if = {
				limit = { has_country_flag = HAI_target_BAH }
				BAH = { country_event = { id = hai_cssm.2 } }
			}
		}
	}
}

HAI_add_ruling_party_popularity = {
	HAI = {
		if = {
			limit = { has_government = democratic }
			add_popularity = { ideology = democratic popularity = 0.03 }
		}
		else_if = {
			limit = { has_government = fascism }
			add_popularity = { ideology = fascism popularity = 0.03 }
		}
		else_if = {
			limit = { has_government = communism }
			add_popularity = { ideology = communism popularity = 0.03 }
		}
		else = {
			add_popularity = { ideology = neutrality popularity = 0.03 }
		}
	}
}

# ROOT is Haiti and FROM is the successful attacker. Dispersed reserve sites
# halve the amount that can be removed in one strike.
HAI_offshore_raid_scaled_fuel_transfer = {
	if = {
		limit = { has_country_flag = HAI_distributed_fuel_reserves }
		if = { limit = { check_variable = { HAI_gonave_oil_stage > 4 } } add_fuel = -15000 FROM = { add_fuel = 15000 } }
		else_if = { limit = { check_variable = { HAI_gonave_oil_stage > 3 } } add_fuel = -12000 FROM = { add_fuel = 12000 } }
		else_if = { limit = { check_variable = { HAI_gonave_oil_stage > 2 } } add_fuel = -9000 FROM = { add_fuel = 9000 } }
		else_if = { limit = { check_variable = { HAI_gonave_oil_stage > 1 } } add_fuel = -6000 FROM = { add_fuel = 6000 } }
		else = { add_fuel = -3000 FROM = { add_fuel = 3000 } }
	}
	else = {
		if = { limit = { check_variable = { HAI_gonave_oil_stage > 4 } } add_fuel = -30000 FROM = { add_fuel = 30000 } }
		else_if = { limit = { check_variable = { HAI_gonave_oil_stage > 3 } } add_fuel = -24000 FROM = { add_fuel = 24000 } }
		else_if = { limit = { check_variable = { HAI_gonave_oil_stage > 2 } } add_fuel = -18000 FROM = { add_fuel = 18000 } }
		else_if = { limit = { check_variable = { HAI_gonave_oil_stage > 1 } } add_fuel = -12000 FROM = { add_fuel = 12000 } }
		else = { add_fuel = -6000 FROM = { add_fuel = 6000 } }
	}
}
