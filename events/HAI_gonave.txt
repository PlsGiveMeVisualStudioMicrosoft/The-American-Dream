# The first-stage raid opens a temporary breach for the attacking gang.
country_event = {
	id = hai_gonave.10
	title = hai_gonave.10.t
	desc = hai_gonave.10.d
	picture = GFX_report_event_generic_political_reform
	is_triggered_only = yes
	option = {
		name = hai_gonave.10.a
		add_manpower = -100
		add_equipment_to_stockpile = { type = infantry_equipment_0 amount = -180 }
		add_stability = -0.02
		custom_effect_tooltip = HAI_escalation_increase_6_tt
		hidden_effect = { add_to_variable = { HAI_escalation = 6 } }
		if = {
			limit = { check_variable = { HAI_gonave_security > 0 } }
			subtract_from_variable = { HAI_gonave_security = 1 }
		}
		if = {
			limit = { has_country_flag = HAI_gonave_tourism_open }
			add_timed_idea = { idea = HAI_gonave_tourism_panic days = 90 }
		}
		else_if = {
			limit = { has_country_flag = HAI_gonave_hotels_built }
			add_timed_idea = { idea = HAI_gonave_construction_abandoned days = 90 }
		}
		FROM = {
			set_country_flag = HAI_gonave_breach
			add_manpower = -100
			add_equipment_to_stockpile = { type = infantry_equipment_0 amount = 120 }
			add_war_support = 0.02
			country_event = { id = hai_gonave.11 hours = 2 }
		}
		HAI_clamp_gonave_security = yes
		HAI_check_escalation_threshold = yes
	}
}
