# Haiti has reached +60 influence. AI partners accept; human partners may refuse.
country_event = {
	id = hai_cssm.1
	title = hai_cssm.1.t
	desc = hai_cssm.1.d
	picture = GFX_report_event_generic_political_reform
	is_triggered_only = yes

	option = {
		name = hai_cssm.1.a
		ai_chance = { factor = 100 }
		add_political_power = 25
		if = {
			limit = { tag = DOM }
			FROM = { set_country_flag = HAI_partner_DOM }
		}
		if = {
			limit = { tag = JAM }
			FROM = { set_country_flag = HAI_partner_JAM }
		}
		if = {
			limit = { tag = BAH }
			FROM = { set_country_flag = HAI_partner_BAH }
		}
		FROM = {
			set_country_flag = HAI_partner_secured
			clr_country_flag = HAI_regional_contest_active
			clr_country_flag = HAI_choose_alternative_partner
			HAI_clear_regional_target_flags = yes
			add_stability = 0.03
			country_event = { id = hai_cssm.3 hours = 3 }
		}
	}

	option = {
		name = hai_cssm.1.b
		ai_chance = { factor = 0 }
		VIV = {
			add_stability = 0.05
			add_war_support = 0.02
		}
		GRF = {
			add_stability = 0.05
			add_war_support = 0.02
		}
		KRB = {
			add_stability = 0.05
			add_war_support = 0.02
		}
		give_military_access = VIV
		give_military_access = GRF
		give_military_access = KRB
		FROM = {
			clr_country_flag = HAI_regional_contest_active
			HAI_clear_regional_target_flags = yes
			if = {
				limit = {
					has_country_flag = HAI_tried_DOM
					has_country_flag = HAI_tried_JAM
					has_country_flag = HAI_tried_BAH
				}
				set_country_flag = HAI_cssm_failed
				country_event = { id = hai_cssm.5 hours = 3 }
			}
			else = {
				set_country_flag = HAI_choose_alternative_partner
				country_event = { id = hai_cssm.4 hours = 3 }
			}
		}
	}
}
